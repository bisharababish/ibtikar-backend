import gradio as gr
from transformers import pipeline
import os

# Load the model - force reload from main branch to get updated config.json
model_name = "Bisharababish/arabert-toxic-classifier"

# Use revision="main" to ensure we get the latest version with fixed config.json
# Also set local_files_only=False to force download if cache is stale
classifier = pipeline(
    "text-classification", 
    model=model_name,
    revision="main",
    model_kwargs={"force_download": False}  # Will still use cache if available, but revision forces main branch
)

def classify(text):
    """Classify text as harmful or safe"""
    result = classifier(text)
    
    # Handle response format
    if isinstance(result, list) and len(result) > 0:
        label = result[0].get("label", "")
        score = result[0].get("score", 0.5)
        
        # LABEL_1 = harmful, LABEL_0 = safe
        if "LABEL_1" in label or label == "1" or "harmful" in label.lower():
            return [{"label": "harmful", "score": float(score)}]
        else:
            return [{"label": "safe", "score": float(score)}]
    
    return [{"label": "unknown", "score": 0.5}]

# Create Gradio interface with API endpoint
iface = gr.Interface(
    fn=classify,
    inputs="text",
    outputs="json",
    api_name="predict",
    title="Arabic Toxic Text Classifier"
)

iface.launch()
