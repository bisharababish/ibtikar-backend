import gradio as gr
from transformers import pipeline, AutoModelForSequenceClassification, AutoTokenizer
import os

# Load the model - handle Git LFS pointers properly
model_name = "Bisharababish/arabert-toxic-classifier"

# Load model and tokenizer separately to handle LFS better
try:
    print("Loading tokenizer...")
    tokenizer = AutoTokenizer.from_pretrained(model_name, revision="main")
    print("Loading model...")
    model = AutoModelForSequenceClassification.from_pretrained(model_name, revision="main")
    print("Creating pipeline...")
    classifier = pipeline("text-classification", model=model, tokenizer=tokenizer)
    print("Model loaded successfully!")
except Exception as e:
    print(f"Error loading model: {e}")
    # Fallback: try with pipeline directly
    print("Trying pipeline directly...")
    classifier = pipeline("text-classification", model=model_name, revision="main")

def classify(text):
    """Classify text as harmful or safe"""
    try:
        result = classifier(text)
        
        # Handle response format
        if isinstance(result, list) and len(result) > 0:
            label = result[0].get("label", "")
            score = result[0].get("score", 0.5)
            
            # LABEL_1 = harmful, LABEL_0 = safe
            if "LABEL_1" in label or label == "1" or "harmful" in label.lower():
                return [{"label": "harmful", "score": float(score)}]
            else:
                return [{"label": "safe", "score": float(score)}]
        
        return [{"label": "unknown", "score": 0.5}]
    except Exception as e:
        print(f"Classification error: {e}")
        return [{"label": "unknown", "score": 0.5}]

# Create Gradio interface with API endpoint
iface = gr.Interface(
    fn=classify,
    inputs="text",
    outputs="json",
    api_name="predict",
    title="Arabic Toxic Text Classifier"
)

iface.launch()
